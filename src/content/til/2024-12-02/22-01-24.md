```rb:_form.html.erb
<%= simple_form_for @user do |f| %>
  <%= f.input :name %>
  <%= f.association :roles, collection: Role.all, input_html: { class: "select2 browser-default", "data-user_role-target": "select", data: { action: "change->user_role#hoge" } } %>
  <%= f.button :submit %>
<% end %>
```
